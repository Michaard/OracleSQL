--[zad5]
	SELECT
		O.id NR,
		C.name KLIENT,
		P.name PRODUKT,
		O.payment_type P£ATNOŒÆ,
		O.date_ordered "DATA ZAMÓWIENIA",
		I.price CENA,
		I.quantity LICZBA
	FROM Ord O, Customer C, Product P, Item I
	WHERE
		O.customer_id=C.id
		AND
		I.ord_id=O.id
		AND
		P.id=I.product_id
		AND
		UPPER(O.payment_type) LIKE 'CASH'
		AND
		TO_CHAR(O.date_ordered, 'YY/MM') LIKE '92/09%'
	ORDER BY O.id, P.name;
  
--[zad6]
	SELECT
		O.id NR,
		C.name KLIENT,
		O.payment_type P£ATNOŒÆ,
		O.date_ordered "DATA ZAMÓWIENIA",
		SUM(I.price*I.quantity) "WYSOKOŒÆ ZAMÓWIENIA"
	FROM Ord O, Customer C, Item I
	WHERE
		O.customer_id=C.id
		AND
		I.ord_id=O.id
		AND
		UPPER(O.payment_type) LIKE 'CASH'
		AND
		TO_CHAR(O.date_ordered, 'YY/MM') LIKE '92/09%'
	GROUP BY C.name, O.id, O.payment_type, O.date_ordered
	ORDER BY O.id;
