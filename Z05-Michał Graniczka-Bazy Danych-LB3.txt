SELECT
  W.city "MIASTO",
  P.name "NAZWA PRODUKTU",
  I1.amount_in_stock "LICZBA PRODUKTÓW"
FROM Warehouse W, Product P, Inventory I1
WHERE
  W.id=I1.warehouse_id
  AND
  P.id=I1.product_id
  AND
  amount_in_stock =
  (SELECT MAX(amount_in_stock)
  FROM Inventory I2
  WHERE I2.warehouse_id=I1.warehouse_id);
